// Konfiguration const OVERPASS_URL = 'https://overpass-api.de/api/interpreter'; const PARK_MIN_CAPACITY = 24; const DEFAULT_PARK_RADIUS = 2000; const DEFAULT_REST_RADIUS = 1000; // Karte initialisieren const map = L.map('map', { zoomControl: true }).setView([51.3388, 6.5853], 14); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap contributors' }).addTo(map); const parkLayer = L.layerGroup().addTo(map); const restLayer = L.layerGroup().addTo(map); const statusEl = document.getElementById('status'); const locateBtn = document.getElementById('locateBtn'); const parkRadiusInput = document.getElementById('parkRadius'); const restRadiusInput = document.getElementById('restRadius'); let currentPos = null; // Hilfsfunktionen function setStatus(txt) { statusEl.textContent = txt; } function createBluePIcon() { const div = document.createElement('div'); div.className = 'blue-p'; div.innerHTML = 'P'; return L.divIcon({ html: div.outerHTML, className: 'leaflet-marker-icon blue-p', iconSize: [28,28] }); } function fetchOverpass(query) { return fetch(OVERPASS_URL, { method: 'POST', body: query, headers: { 'Content-Type': 'text/plain' } }).then(r => r.json()); } function buildParkingQuery(lat, lon, radius) { // Wir holen nodes, ways und relations mit amenity=parking oder parking=surface etc. return `[out:json][timeout:25]; ( node(around:${radius},${lat},${lon})[amenity=parking]; way(around:${radius},${lat},${lon})[amenity=parking]; relation(around:${radius},${lat},${lon})[amenity=parking]; ); out center tags;`; } function buildRestaurantQuery(lat, lon, radius) { return `[out:json][timeout:25]; ( node(around:${radius},${lat},${lon})[amenity=restaurant]; way(around:${radius},${lat},${lon})[amenity=restaurant]; relation(around:${radius},${lat},${lon})[amenity=restaurant]; ); out center tags;`; } function parseCapacity(cap) { if (!cap) return null; const n = parseInt(cap, 10); return isNaN(n) ? null : n; } function isPaid(tags) { if (!tags) return false; if (tags.fee && tags.fee.toLowerCase() === 'yes') return true; if (tags['fee:conditional'] && tags['fee:conditional'].toLowerCase().includes('yes')) return true; if (tags['parking:fee'] && tags['parking:fee'].toLowerCase() === 'yes') return true; return false; } function showParkings(osmJson) { parkLayer.clearLayers(); const icon = createBluePIcon(); let count = 0; osmJson.elements.forEach(el => { const tags = el.tags || {}; if (isPaid(tags)) return; const cap = parseCapacity(tags.capacity); if (cap !== null && cap < PARK_MIN_CAPACITY) return; // Wenn capacity fehlt, optional: ausschließen oder anzeigen. Hier zeigen wir nur mit capacity oder großen Flächen. if (cap === null) return; const lat = el.lat || (el.center && el.center.lat); const lon = el.lon || (el.center && el.center.lon); if (!lat || !lon) return; const marker = L.marker([lat, lon], { icon }); const name = tags.name || 'Parkplatz'; marker.bindPopup(`${name}
Kapazität: ${cap}
${tags.fee ? 'Gebührenpflichtig' : 'Kostenlos'}`); marker.addTo(parkLayer); count++; }); setStatus(`Gefundene Parkplätze (≥ ${PARK_MIN_CAPACITY}): ${count}`); } function showRestaurants(osmJson) { restLayer.clearLayers(); let count = 0; osmJson.elements.forEach(el => { const tags = el.tags || {}; const lat = el.lat || (el.center && el.center.lat); const lon = el.lon || (el.center && el.center.lon); if (!lat || !lon) return; const oh = tags.opening_hours; let openSatAt15 = false; if (oh) { try { const ohObj = new opening_hours(oh); // Prüfen ob am nächsten Samstag um 15:00 geöffnet ist const now = new Date(); // Finde nächsten Samstag const saturday = new Date(now); saturday.setDate(now.getDate() + ((6 - now.getDay() + 7) % 7)); saturday.setHours(15,0,0,0); openSatAt15 = ohObj.getState(saturday); // Wenn opening_hours nicht eindeutig ist, ohObj.getNextChange kann helfen } catch (e) { // falls parsing fehlschlägt, überspringen return; } } else { // Ohne opening_hours nicht anzeigen return; } if (!openSatAt15) return; // Ausschluss: Lokale die erst ab 17:00 öffnen // Wir prüfen, ob das erste Öffnen am Samstag später als 15:00 ist. try { const ohObj = new opening_hours(tags.opening_hours); const saturday = new Date(); saturday.setDate(new Date().getDate() + ((6 - new Date().getDay() + 7) % 7)); saturday.setHours(0,0,0,0); const intervals = ohObj.getOpenIntervals(saturday, new Date(saturday.getTime() + 24*3600*1000)); if (!intervals || intervals.length === 0) return; const firstOpen = new Date(intervals[0][0]); if (firstOpen.getHours() >= 17) return; } catch (e) { return; } const marker = L.marker([lat, lon]); const name = tags.name || 'Restaurant'; marker.bindPopup(`${name}
Öffnungszeiten: ${tags.opening_hours}`); marker.addTo(restLayer); count++; }); setStatus(prev => prev + ` | Restaurants offen Sa ab 15:00: ${count}`); } function queryAndShow() { if (!currentPos) return; const lat = currentPos.coords.latitude; const lon = currentPos.coords.longitude; const parkRadius = parseInt(parkRadiusInput.value, 10) || DEFAULT_PARK_RADIUS; const restRadius = parseInt(restRadiusInput.value, 10) || DEFAULT_REST_RADIUS; setStatus('Suche Parkplätze...'); const parkQ = buildParkingQuery(lat, lon, parkRadius); fetchOverpass(parkQ).then(json => { showParkings(json); }).catch(err => setStatus('Fehler bei Parkabfrage')); setStatus('Suche Restaurants...'); const restQ = buildRestaurantQuery(lat, lon, restRadius); fetchOverpass(restQ).then(json => { showRestaurants(json); }).catch(err => setStatus('Fehler bei Restaurantabfrage')); } // Standortfunktionen function onLocationFound(e) { currentPos = e; const lat = e.coords.latitude; const lon = e.coords.longitude; map.setView([lat, lon], 16); L.circle([lat, lon], { radius: Math.max(e.coords.accuracy, 10), color:'#3388ff', fill:false }).addTo(map); queryAndShow(); } function onLocationError(err) { setStatus('Standort nicht verfügbar: ' + err.message); } // Initiale Standortanfrage locateBtn.addEventListener('click', () => { setStatus('Standort wird ermittelt...'); navigator.geolocation.getCurrentPosition(pos => { onLocationFound({ coords: pos.coords }); }, onLocationError, { enableHighAccuracy: true, maximumAge: 10000, timeout: 10000 }); }); // Auto zentrieren beim Laden if (navigator.geolocation) { setStatus('Standort wird ermittelt...'); navigator.geolocation.getCurrentPosition(pos => { onLocationFound({ coords: pos.coords }); }, onLocationError, { enableHighAccuracy: true, maximumAge: 10000, timeout: 10000 }); } else { setStatus('Geolocation nicht unterstützt'); } // Neu laden bei Radiusänderung parkRadiusInput.addEventListener('change', queryAndShow); restRadiusInput.addEventListener('change', queryAndShow);
